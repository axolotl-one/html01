function Node(value){
  this.left = null;
  this.right = null;
  this.value = value;
}

function inorder(node){
  if(node === null) return;
  inorder(node.left);
  console.log(node.value);
  inorder(node.right);
}

function preorder(node){
  if(node === null) return;
  console.log(node.value);
  preorder(node.left);
  preorder(node.right);
}

function postorder(node){
  if(node === null) return;
  postorder(node.left);
  postorder(node.right);
  console.log(node.value);
}
