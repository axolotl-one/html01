Estructuras de Control

Condicional if - elif - else

 
if condicion1:
     # bloque de instrucciones si se cumple condicion1
elif condicion2:
     # bloque de instrucciones si se cumple condicion2
elif condicionN:
     # bloque de instrucciones si se cumple condicionN
else:
     # bloque de instrucciones si no se cumplió ninguna de las condiciones anteriores

Ciclo for (foreach)

 
for letra in palabra:
     # similar a foreach(letra in palabra) en C#
     print(letra) # trabaja con cada letra de una palabra, siendo letra cada elemento de una lista

Ciclo for - range (for tradicional)

 
for i in range(inicial, limite, paso):
     # similar a for(int i = inicial; i < limite; paso)
     print(i)

Ciclo for - range (for con repeticiones establecidas)

 
for i in range(nRepeticiones):
     print(nRepeticiones)

Ciclo while (while tradicional)

 
while condicion:
     # bloque de instrucciones, la condicion debe ser falsa en algún punto para cerrar el ciclo

Ciclo while (do-while)

 
while true:
     # bloque de instrucciones
     if(condicionCierre)
          break      # Si se cumple la condición de cierre, termina el ciclo con "break"

Sin parametros

def nombreFuncion(var1, var2, var3, ..., varN):

     # Ejemplo: print("Primer variable: " + var1 + ". Segunda variable: " + var2 + "...")
Con parametros prestablecidos 
def nombreFuncion(varSinPrestablecer, varPrestablecida1 = dato1, varPrestablecidaN = datoN):
     # Nota: las variables prestablecidas deben definirse después de las variables no prestablecidas
     # bloque de código
 
# Ejemplo:
def imprimirVariables(var1, varPrest1 = "rueda", varPrest2 = 10):
     print("Primera: " + var1 + ". Segunda: " varPrest1 + ". Tercera: " + varPrest2)"
 
imprimirVariables("prueba", "rodillo", 21)  # Imprime: Primera: prueba. Segunda: rodillo. Tercera: 21
imprimirVariables("estado")                 # Imprime: Primera: estado. Segunda: rueda. Tercera: 10
Con cantidad indefenida de parametros no prestablecidos 
def nombreFuncion(*args):
     # bloque de código
 
# Ejemplo:
def ImprimirNParametros(*args):
     for parametro in args:
          print(parametro)
 
ImprimirNParametros(2, "fierro", 2.4, "HDMI")   # Imprime: 2 fierro 2.4 HDMI
Con cantidad indefenida de parametros con variable asignada 
def nombreFuncion(**kwargs):
     # bloque de código
 
# Ejemplo:
def ImprimirNParametros(**kwargs):
     for llave in kwargs:
          print(kwargs[llave])
 
ImprimirNParametros(a=2, f="fierro", dr=2.4, w="HDMI")   # Imprime: 2 fierro 2.4 HDMI

Clases
Estructura Básica 
class nombreClase():
     # Creación del constructor
     def __init__(self, param1, param2, ..., paramN):
          self.param1 = param1
          self.param2 = param2
          # ...
          self.paramN = paramN
 
     def metodo01(self):
          print("Mostrar param1 (cadena string): " + param1)
          print("Mostrar param2 (numero): + str(param2))
 
# Crear Objeto de Clase:
nombreObjeto = nombreCLase(valor1, valor2, ..., valorN)
 
# Acceso a Propiedades
print(nombreObjeto.param1)    # Obtiene la propiedad "param1" del objeto "nombreObjeto y la imprime"
nombreObjeto.param1 = valor   # Modifica (reasigna) un "valor" a la propiedad "param1" de "nombreObjeto"
# Acceso a Métodos
nombreObjeto.metodo01()
Estructura de Herencia 
class claseHijo(clasePadre):
     # Definir Constructor
     def __init__(self, paramPadre1, paramPadre2, ..., paramPadreN, param1,  param2, ..., paramN)
          # Construir propiedades a partir de la clase padre
          super().__init__(paramPadre1, paramPadre2, ..., paramPadreN)
          # Construir propiedades propias de la clase
          self.param1 = param1
          self.param2 = param2
          ...
          self.paramN = paramN
 
     metodoPadre(self):
          # Método de la clase padre modificado en esta clase (polimorfismo)
          # Se llama al método de la clase padre por su nombre y se sobreescribe el bloque de código
